```bash
pnpx shadcn@latest add https://ui.dninomiya.com/r/use-is-active-path.json
```

## 使用法

```tsx
import { useIsActivePath } from "@/hooks/use-is-active-path";

const isActivePath = useIsActivePath();

// パスの前方一致で判定
const isActive = isActivePath("/dashboard");

// 完全一致で判定
const isExactActive = isActivePath("/dashboard", { exact: true });
```
